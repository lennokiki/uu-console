          function logAllTheme() {
            const themes = [];
            const prefix = 'No';
            for (let i = 1; i <= 9; i++) {
              for (let j = 1; j <= 6; j++) {
                themes.push(`${prefix}${i}0${j}`);
              }
            }
            let i = 0;
            themes.push('Nox01');
            uuConsole.useSize('small');
            themes.forEach(item => {
              uuConsole.useTheme(item).log('Hi', 'UU', 'Console', '---' + item);
            });
          }

custom theme

          function logCustomTheme() {
            const themeProps = [
              {
                color: '#fff',
                background: '#ff00ee',
                fontWeight: 'bold',
              },
              {
                color: '#555',
                background: '#dd33aa',
                fontStyle: 'italic',
                letterSpacing: '3px',
              },
              'color: #ff00cc; background: #111;',
            ];
            uuConsole
              .setCustomTheme('my-theme', themeProps)
              .useTheme('my-theme')
              .useSize('small')
              .log('Hello', 'B-O-Y,', 'Where are you from');
          }

del custom theme

          function deleteCustomTheme() {
            uuConsole.deleteCustomTheme('my-theme');
          }
